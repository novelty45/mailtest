from js import document def process_input(event): # 사용자 입력 값 가져오기 user_input = document.getElementById("userInput").value # 입력값을 대문자로 변환 processed_input = user_input.upper() # 결과를 처리하여 표시 result_element = document.getElementById("result") result_element.innerHTML = f"Processed input: {processed_input}" # JavaScript 이벤트 리스너 설정 submit_button = document.getElementById("submitButton") submit_button.addEventListener("click", process_input)
